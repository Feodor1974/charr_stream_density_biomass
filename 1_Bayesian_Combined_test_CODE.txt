"""
Bayesian_RealData_W1param_M0gtW1_v2.py
Reparameterization: W1 ~ Uniform (hard bounds), beta1 deterministic
2 parameters: beta0, W1
lambda0 = m0/N0 (fixed), lambda1 = m1/(N0*beta0), beta1 = 1 - W1 - lambda1, M0 = 1 - beta0 - lambda0
Parameters grouped by age: 0+ first (beta0, M0), then 1+ (W1, beta1, lambda1)
lambda0 is fixed (not in model), added to Excel and sum check manually
"""

import pandas as pd
import numpy as np
import pymc as pm
import arviz as az
import matplotlib.pyplot as plt
from scipy.optimize import differential_evolution
import warnings
warnings.filterwarnings('ignore')

# ============================================================
# 1. READ DATA
# ============================================================

df = pd.read_excel('Combined_M.xlsx', header=0)
df = df.set_index('Parameters')

params = {}
for morph in ['Nosed', 'White']:
    params[morph] = {
        'm0': df.loc['lambda_Age 0+', morph],
        'm1': df.loc['lambda_Age 1+', morph],
        'm2': df.loc['lambda_Age 2+', morph],
        'M_min': df.loc['M_min_0+', morph],
        'M_max': df.loc['M_max_0+', morph],
        'W_min': df.loc['W\u2081_min_1+', morph],
        'W_max': df.loc['W\u2081_max_1+', morph],
        'n_spawn_min': df.loc['n_spawners_min', morph],
        'n_spawn_max': df.loc['n_spawners_max', morph],
        'fec_min': df.loc['fecundity_min', morph],
        'fec_max': df.loc['fecundity_max', morph],
        'egg_mort_min': df.loc['egg_mortality_min', morph],
        'egg_mort_max': df.loc['egg_mortality_max', morph],
    }

for morph in ['Nosed', 'White']:
    p = params[morph]
    n_spawn = (p['n_spawn_min'] + p['n_spawn_max']) / 2
    fec = (p['fec_min'] + p['fec_max']) / 2
    egg_surv = 1 - (p['egg_mort_min'] + p['egg_mort_max']) / 2
    p['N0_base'] = n_spawn * fec * egg_surv
    p['lambda0'] = p['m0'] / p['N0_base']

print("=== BASE DATA ===")
for morph in ['Nosed', 'White']:
    p = params[morph]
    print(f"\n{morph}: N0={p['N0_base']:.1f}, lambda0={p['lambda0']:.4f}")
    print(f"  m0={p['m0']:.0f}, m1={p['m1']:.0f}, m2={p['m2']:.0f}")

# ============================================================
# 2. BOUND OPTIMIZATION
# ============================================================

def level1_objective(bounds, params, delta=0.05):
    M_min, M_max, W_min, W_max = bounds
    if M_min >= M_max or W_min >= W_max or M_min <= W_max + delta:
        return 1e10
    total_obj = 0
    for morph in ['Nosed', 'White']:
        p = params[morph]
        lambda0 = p['m0'] / p['N0_base']
        beta0_min = max(0.001, 1 - M_max - lambda0)
        beta0_max = 1 - M_min - lambda0
        if beta0_min >= beta0_max:
            total_obj += 1e6
            continue
        beta0_test = np.linspace(beta0_min, beta0_max, 1000)
        lambda1_vals = p['m1'] / (p['N0_base'] * beta0_test)
        beta1_vals = p['m2'] / (p['N0_base'] * beta0_test)
        W1_vals = 1 - beta1_vals - lambda1_vals
        M0_vals = 1 - beta0_test - lambda0
        feasible = (W1_vals >= W_min) & (W1_vals <= W_max) & (M0_vals > W1_vals + delta)
        if not np.any(feasible):
            total_obj += 1e6
            continue
        idx = np.where(feasible)[0][len(np.where(feasible)[0])//2]
        beta0 = beta0_test[idx]
        lambda1 = p['m1'] / (p['N0_base'] * beta0)
        beta1 = p['m2'] / (p['N0_base'] * beta0)
        m1_model = p['N0_base'] * beta0 * lambda1
        m2_model = p['N0_base'] * beta0 * beta1
        total_obj += ((m1_model - p['m1'])/p['m1'])**2 + ((m2_model - p['m2'])/p['m2'])**2
    return total_obj

result_level1 = differential_evolution(
    level1_objective, [(0.05,0.5), (0.3,0.95), (0.05,0.5), (0.05,0.5)],
    args=(params, 0.05), seed=42, maxiter=300, tol=1e-8, polish=True
)

if not hasattr(result_level1, 'x') or len(result_level1.x) < 4:
    print("Optimization failed, using fallback bounds")
    M_min_opt, M_max_opt = 0.30, 0.70
    W1_min_opt, W1_max_opt = 0.30, 0.70
else:
    M_min_opt, M_max_opt, W1_min_opt, W1_max_opt = result_level1.x

print(f"\nCommon bounds: M0=[{M_min_opt:.3f}, {M_max_opt:.3f}], W1=[{W1_min_opt:.3f}, {W1_max_opt:.3f}]")

# ============================================================
# 3. BAYESIAN MODEL: W1 ~ UNIFORM (HARD BOUNDS)
# ============================================================

all_traces = {}
all_summaries = {}

for morph in ['Nosed', 'White']:
    p = params[morph]
    N0_fixed = p['N0_base']
    lambda0 = p['lambda0']

    beta0_min = max(0.001, 1 - M_max_opt - lambda0)
    beta0_max = 1 - M_min_opt - lambda0

    W1_min = W1_min_opt
    W1_max = W1_max_opt

    print(f"\n=== {morph} ===")
    print(f"beta0: [{beta0_min:.4f}, {beta0_max:.4f}]")
    print(f"W1: [{W1_min:.4f}, {W1_max:.4f}]")

    with pm.Model() as model:
        # 2 free parameters
        beta0 = pm.Uniform('beta0', lower=beta0_min, upper=beta0_max)
        W1 = pm.Uniform('W1', lower=W1_min, upper=W1_max)

        # Deterministic — grouped by age: 0+ first, then 1+
        M0 = pm.Deterministic('M0', 1 - beta0 - lambda0)
        lambda1 = pm.Deterministic('lambda1', p['m1'] / (N0_fixed * beta0))
        beta1 = pm.Deterministic('beta1', 1 - W1 - lambda1)

        # Likelihood m2: N0*beta0*beta1 = m2
        pred_m2 = N0_fixed * beta0 * beta1
        pm.Normal('likelihood_m2', mu=pred_m2, sigma=0.05 * p['m2'], observed=p['m2'])

        # Pseudo-likelihood for stability
        pm.Normal('pseudo_m2', mu=pred_m2 - p['m2'], sigma=16000, observed=0)

        # M0 > W1 + 0.05
        pm.Potential('M0_gt_W1', -10000 * pm.math.switch(M0 > W1 + 0.05, 0, (W1 + 0.05 - M0)**2))

        trace = pm.sample(4000, tune=2000, chains=4, cores=4,
                         return_inferencedata=True,
                         target_accept=0.95,
                         random_seed=42)

    all_traces[morph] = trace

    # Summary grouped by age: 0+ first (beta0, M0), then 1+ (W1, beta1, lambda1)
    summary = az.summary(trace, var_names=['beta0', 'M0', 'W1', 'beta1', 'lambda1'],
                         hdi_prob=0.95)
    all_summaries[morph] = summary

    print(f"\n--- Posterior Summary ---")
    print(summary.to_string())

    rhat_max = summary['r_hat'].max()
    ess_min = summary['ess_bulk'].min()
    divergences = trace.sample_stats.diverging.sum().values
    print(f"\nDiagnostics: R-hat(max)={rhat_max:.3f}, ESS(min)={ess_min:.0f}, Divergences={divergences}")

# ============================================================
# 4. PLOTS (grouped by age)
# ============================================================

print("\n=== PLOTS ===")

for morph in ['Nosed', 'White']:
    trace = all_traces[morph]

    age0_vars = ['beta0', 'M0']
    age1_vars = ['W1', 'beta1', 'lambda1']
    all_vars = age0_vars + age1_vars

    az.plot_trace(trace, var_names=all_vars,
                  compact=True, figsize=(12, 10))
    plt.suptitle(f'{morph} — Trace & Posterior (Age 0+ then Age 1+)', fontsize=14)
    plt.tight_layout()
    plt.savefig(f'BayesianW1_{morph}_TracePlot_v2.png', dpi=150, bbox_inches='tight')
    plt.show()

    az.plot_pair(trace, var_names=all_vars,
                 kind='kde', marginals=True, figsize=(10, 10))
    plt.suptitle(f'{morph} — Pair Plot (Age 0+ then Age 1+)', fontsize=14)
    plt.tight_layout()
    plt.savefig(f'BayesianW1_{morph}_PairPlot_v2.png', dpi=150, bbox_inches='tight')
    plt.show()

    az.plot_forest(trace, var_names=all_vars,
                   combined=True, hdi_prob=0.95, figsize=(8, 6))
    plt.suptitle(f'{morph} — 95% HDI (Age 0+ then Age 1+)', fontsize=14)
    plt.tight_layout()
    plt.savefig(f'BayesianW1_{morph}_ForestPlot_v2.png', dpi=150, bbox_inches='tight')
    plt.show()

# ============================================================
# 5. SUMMARY TABLE (grouped by age, lambda0 added manually)
# ============================================================

rows = []
for morph in ['Nosed', 'White']:
    summary = all_summaries[morph]
    p = params[morph]
    lambda0_val = p['lambda0']

    # Age 0+: beta0, lambda0 (fixed), M0
    for param in ['beta0', 'M0']:
        row = summary.loc[param]
        rows.append({
            'Morph': morph, 'Parameter': param,
            'Mean': row['mean'], 'SD': row['sd'],
            'HDI_2.5%': row['hdi_2.5%'], 'HDI_97.5%': row['hdi_97.5%'],
            'ESS_bulk': row['ess_bulk'], 'R_hat': row['r_hat']
        })
    # lambda0 as fixed value
    rows.append({
        'Morph': morph, 'Parameter': 'lambda0',
        'Mean': lambda0_val, 'SD': 0.0,
        'HDI_2.5%': lambda0_val, 'HDI_97.5%': lambda0_val,
        'ESS_bulk': np.nan, 'R_hat': np.nan
    })

    # Age 1+: W1, beta1, lambda1
    for param in ['W1', 'beta1', 'lambda1']:
        row = summary.loc[param]
        rows.append({
            'Morph': morph, 'Parameter': param,
            'Mean': row['mean'], 'SD': row['sd'],
            'HDI_2.5%': row['hdi_2.5%'], 'HDI_97.5%': row['hdi_97.5%'],
            'ESS_bulk': row['ess_bulk'], 'R_hat': row['r_hat']
        })

df_results = pd.DataFrame(rows)
df_results.to_excel('BayesianW1_Results_Summary_v2.xlsx', index=False)

print("\n=== DONE ===")
print("Saved: BayesianW1_Results_Summary_v2.xlsx")
