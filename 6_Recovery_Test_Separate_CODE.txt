
"""
Recovery_Test_Separate_M0gtW1.py
Раздельная смертность: recovery test для 3 свободных параметров (beta0, W1, P0)
Deterministic: W0 = 1 - beta0 - lambda0 - P0, M0 = P0 + W0, lambda1, beta1
"""

import pandas as pd
import numpy as np
import pymc as pm
import arviz as az
from scipy.optimize import differential_evolution
import warnings
warnings.filterwarnings('ignore')

# ============================================================
# 1. ЧТЕНИЕ ДАННЫХ
# ============================================================

df = pd.read_excel('Separate_M.xlsx', header=0)
df = df.set_index('Parameters')

params = {}
for morph in ['Nosed', 'White']:
    params[morph] = {
        'm0': df.loc['lambda_Age 0+', morph],
        'm1': df.loc['lambda_Age 1+', morph],
        'm2': df.loc['lambda_Age 2+', morph],
        'P_min': df.loc['P₀_min', morph],
        'P_max': df.loc['P₀_max', morph],
        'W0_min': df.loc['W₀_min', morph],
        'W0_max': df.loc['W₀_max', morph],
        'W1_min': df.loc['W₁_min', morph],
        'W1_max': df.loc['W₁_max', morph],
        'n_spawn_min': df.loc['n_spawners_min', morph],
        'n_spawn_max': df.loc['n_spawners_max', morph],
        'fec_min': df.loc['fecundity_min', morph],
        'fec_max': df.loc['fecundity_max', morph],
        'egg_mort_min': df.loc['egg_mortality_min', morph],
        'egg_mort_max': df.loc['egg_mortality_max', morph],
    }

for morph in ['Nosed', 'White']:
    p = params[morph]
    n_spawn = (p['n_spawn_min'] + p['n_spawn_max']) / 2
    fec = (p['fec_min'] + p['fec_max']) / 2
    egg_surv = 1 - (p['egg_mort_min'] + p['egg_mort_max']) / 2
    p['N0_base'] = n_spawn * fec * egg_surv
    p['lambda0'] = p['m0'] / p['N0_base']

# ============================================================
# 2. ОПТИМИЗАЦИЯ ГРАНИЦ
# ============================================================

def level1_objective(bounds, params, delta=0.05):
    M_min, M_max, W1_min, W1_max = bounds
    if M_min >= M_max or W1_min >= W1_max or M_min <= W1_max + delta:
        return 1e10
    total_obj = 0
    for morph in ['Nosed', 'White']:
        p = params[morph]
        lambda0 = p['lambda0']
        beta0_min = max(0.001, 1 - M_max - lambda0)
        beta0_max = 1 - M_min - lambda0
        if beta0_min >= beta0_max:
            total_obj += 1e6
            continue
        beta0_test = np.linspace(beta0_min, beta0_max, 1000)
        lambda1_vals = p['m1'] / (p['N0_base'] * beta0_test)
        beta1_vals = p['m2'] / (p['N0_base'] * beta0_test)
        W1_vals = 1 - beta1_vals - lambda1_vals
        M0_vals = 1 - beta0_test - lambda0
        feasible = (W1_vals >= W1_min) & (W1_vals <= W1_max) & (M0_vals > W1_vals + delta)
        for i in range(len(beta0_test)):
            if not feasible[i]:
                continue
            M0_i = M0_vals[i]
            P0_min_feas = max(p['P_min'], M0_i - p['W0_max'])
            P0_max_feas = min(p['P_max'], M0_i - p['W0_min'])
            if P0_min_feas > P0_max_feas:
                feasible[i] = False
        if not np.any(feasible):
            total_obj += 1e6
            continue
        idx = np.where(feasible)[0][len(np.where(feasible)[0])//2]
        beta0 = beta0_test[idx]
        lambda1 = p['m1'] / (p['N0_base'] * beta0)
        beta1 = p['m2'] / (p['N0_base'] * beta0)
        m1_model = p['N0_base'] * beta0 * lambda1
        m2_model = p['N0_base'] * beta0 * beta1
        total_obj += ((m1_model - p['m1'])/p['m1'])**2 + ((m2_model - p['m2'])/p['m2'])**2
    return total_obj

result_level1 = differential_evolution(
    level1_objective, [(0.05, 0.5), (0.3, 0.95), (0.05, 0.5), (0.05, 0.5)],
    args=(params, 0.05), seed=42, maxiter=300, tol=1e-8, polish=True
)
M_min_opt, M_max_opt, W1_min_opt, W1_max_opt = result_level1.x

print(f"Общие границы: M0=[{M_min_opt:.3f}, {M_max_opt:.3f}], W1=[{W1_min_opt:.3f}, {W1_max_opt:.3f}]")

# ============================================================
# 3. ГЕНЕРАЦИЯ ИСТИННЫХ ПАРАМЕТРОВ
# ============================================================

def generate_true_params(p, M_min, M_max, W1_min, W1_max, delta=0.05):
    lambda0 = p['lambda0']
    beta0_min = max(0.001, 1 - M_max - lambda0)
    beta0_max = 1 - M_min - lambda0

    beta0_true = np.random.uniform(beta0_min, beta0_max)
    lambda1_true = p['m1'] / (p['N0_base'] * beta0_true)
    beta1_true = p['m2'] / (p['N0_base'] * beta0_true)
    W1_true = 1 - beta1_true - lambda1_true
    M0_true = 1 - beta0_true - lambda0

    # P0 и W0: P0 in [P_min, P_max], W0 = M0 - P0 in [W0_min, W0_max]
    P0_min_feas = max(p['P_min'], M0_true - p['W0_max'])
    P0_max_feas = min(p['P_max'], M0_true - p['W0_min'])
    if P0_min_feas >= P0_max_feas:
        # Fallback: берём центр feasible beta0
        beta0_test = np.linspace(beta0_min, beta0_max, 1000)
        lambda1_vals = p['m1'] / (p['N0_base'] * beta0_test)
        beta1_vals = p['m2'] / (p['N0_base'] * beta0_test)
        W1_vals = 1 - beta1_vals - lambda1_vals
        M0_vals = 1 - beta0_test - lambda0
        feasible = (W1_vals >= W1_min) & (W1_vals <= W1_max) & (M0_vals > W1_vals + delta)
        for i in range(len(beta0_test)):
            if not feasible[i]:
                continue
            M0_i = M0_vals[i]
            P0_min_i = max(p['P_min'], M0_i - p['W0_max'])
            P0_max_i = min(p['P_max'], M0_i - p['W0_min'])
            if P0_min_i > P0_max_i:
                feasible[i] = False
        if np.any(feasible):
            idx = np.where(feasible)[0][len(np.where(feasible)[0])//2]
            beta0_true = beta0_test[idx]
            lambda1_true = p['m1'] / (p['N0_base'] * beta0_true)
            beta1_true = p['m2'] / (p['N0_base'] * beta0_true)
            W1_true = 1 - beta1_true - lambda1_true
            M0_true = 1 - beta0_true - lambda0
            P0_min_feas = max(p['P_min'], M0_true - p['W0_max'])
            P0_max_feas = min(p['P_max'], M0_true - p['W0_min'])

    P0_true = np.random.uniform(P0_min_feas, P0_max_feas)
    W0_true = M0_true - P0_true

    m2_true = p['N0_base'] * beta0_true * beta1_true

    return {
        'beta0': beta0_true, 'W1': W1_true, 'P0': P0_true,
        'W0': W0_true, 'M0': M0_true,
        'beta1': beta1_true, 'lambda1': lambda1_true,
        'm2': m2_true
    }

# ============================================================
# 4. RECOVERY TEST (только свободные параметры: beta0, W1, P0)
# ============================================================

def run_recovery_test(morph, p, M_min, M_max, W1_min, W1_max,
                       n_sims=20, noise_frac=0.15, delta=0.05):
    results = []

    print(f"\n=== {morph} RECOVERY TEST ({n_sims} sims) ===")
    print("Проверяются свободные параметры: beta0, W1, P0")
    print("Deterministic (W0, M0, beta1, lambda1) не проверяются отдельно\n")

    for sim in range(n_sims):
        true = generate_true_params(p, M_min, M_max, W1_min, W1_max, delta)

        m1_obs = p['m1']  # exact
        m2_obs = max(np.random.normal(true['m2'], noise_frac * true['m2']), 10.0)

        lambda0 = p['lambda0']
        beta0_min = max(0.001, 1 - M_max - lambda0)
        beta0_max = 1 - M_min - lambda0

        # P0 bounds (worst-case)
        P0_min_overall = p['P_min']
        P0_max_overall = min(p['P_max'], 1 - beta0_min - lambda0 - p['W0_min'])
        P0_min_overall = max(P0_min_overall, 1 - beta0_max - lambda0 - p['W0_max'])
        if P0_min_overall >= P0_max_overall:
            P0_min_overall = p['P_min']
            P0_max_overall = p['P_max']

        with pm.Model() as model:
            beta0 = pm.Uniform('beta0', lower=beta0_min, upper=beta0_max)
            W1 = pm.Uniform('W1', lower=W1_min, upper=W1_max)
            P0 = pm.Uniform('P0', lower=P0_min_overall, upper=P0_max_overall)

            W0 = pm.Deterministic('W0', 1 - beta0 - lambda0 - P0)
            M0 = pm.Deterministic('M0', P0 + W0)
            lambda1 = pm.Deterministic('lambda1', p['m1'] / (p['N0_base'] * beta0))
            beta1 = pm.Deterministic('beta1', 1 - W1 - lambda1)

            pred_m2 = p['N0_base'] * beta0 * beta1
            pm.Normal('likelihood_m2', mu=pred_m2, sigma=0.05 * p['m2'], observed=m2_obs)

            pm.Potential('M0_gt_W1', -10000 * pm.math.switch(M0 > W1 + delta, 0, (W1 + delta - M0)**2))
            pm.Potential('W0_positive', -10000 * pm.math.switch(W0 > 0.001, 0, (0.001 - W0)**2))
            pm.Potential('W0_in_bounds', -10000 * (pm.math.switch(W0 >= p['W0_min'], 0, (p['W0_min'] - W0)**2) +
                                                    pm.math.switch(W0 <= p['W0_max'], 0, (W0 - p['W0_max'])**2)))

            trace = pm.sample(2000, tune=1000, chains=4, cores=4,
                             return_inferencedata=True,
                             target_accept=0.95,
                             random_seed=42+sim)

        summary = az.summary(trace, var_names=['beta0', 'W1', 'P0'],
                             hdi_prob=0.95)
        rhat_max = summary['r_hat'].max()
        ess_min = summary['ess_bulk'].min()
        divergences = trace.sample_stats.diverging.sum().values

        # Recovery: только свободные параметры
        recovered = {}
        for param in ['beta0', 'W1', 'P0']:
            hdi = az.hdi(trace, var_names=[param], hdi_prob=0.95)
            true_val = true[param]
            lower = hdi[param].sel(hdi='lower').values
            higher = hdi[param].sel(hdi='higher').values
            recovered[param] = (lower <= true_val <= higher)

        all_rec = all(recovered.values())
        status = 'OK' if rhat_max < 1.01 and divergences == 0 and ess_min > 400 else 'WARN'

        print(f"  Sim {sim+1}/{n_sims}: "
              f"true(b0={true['beta0']:.3f}, W1={true['W1']:.3f}, P0={true['P0']:.3f}), "
              f"m2_obs={m2_obs:.0f}, "
              f"R-hat={rhat_max:.3f}, ESS={ess_min:.0f}, Div={divergences}, "
              f"rec={sum(recovered.values())}/3, {status}")

        results.append({
            'sim': sim+1, 'm2_obs': m2_obs,
            'beta0_true': true['beta0'], 'W1_true': true['W1'], 'P0_true': true['P0'],
            'W0_true': true['W0'], 'M0_true': true['M0'],
            'beta1_true': true['beta1'], 'lambda1_true': true['lambda1'],
            'rhat_max': rhat_max, 'ess_min': ess_min, 'divergences': divergences,
            'rec_beta0': recovered['beta0'], 'rec_W1': recovered['W1'], 'rec_P0': recovered['P0'],
            'all_recovered': all_rec,
            'status': status
        })

    return results

all_results = {}
for morph in ['Nosed', 'White']:
    res = run_recovery_test(morph, params[morph], M_min_opt, M_max_opt,
                           W1_min_opt, W1_max_opt, n_sims=20, noise_frac=0.15)
    all_results[morph] = res

# ============================================================

                     f"m2_obs={r['m2_obs']:.0f}|rhat={r['rhat_max']:.3f}|"
                     f"ess={r['ess_min']:.0f}|div={r['divergences']}|"
                     f"rec_b0={r['rec_beta0']}|rec_W1={r['rec_W1']}|rec_P0={r['rec_P0']}|"
                     f"status={r['status']}"
        })

df_results = pd.DataFrame(rows)
